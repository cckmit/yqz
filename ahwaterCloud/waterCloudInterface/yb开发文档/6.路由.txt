路由
	列出所有租户路由（用于管理员界面）
		url:"contr/ListAllRouter"
		输出字段：
					{
						id(routerId)(隐含字段，不显示)
						名称（routerName）
						状态（state）
						外部网络(externalNetworkName)
						管理员状态(adminState)
						项目名(projectName)
					}
	列出当前租户所有路由
		url:"contr/ListTenantRouter"
		输出字段：
					{
						id(routerId)(隐含字段，不显示)
						名称（routerName）
						状态（state）
						外部网络(externalNetworkName)
						管理员状态(adminState)
					}
						
	详情：(概况)
		url:"/contr/GetRouterDetail"
		输入字段：
					id(routerId)
		输出字段：
					{
						名称（routerName）
						id(routerId)
						项目id(projectId)
						状态(state)
						管理员状态(adminState)
						网络名称(externalNetworkName)
						网络ID(externalNetworkId)
						SNAT(snat)
					}
